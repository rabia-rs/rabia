72    pub async fn run(mut self) -> Result<()> {
73        info!("Starting Rabia consensus engine for node {}", self.node_id);
74
75        let mut cleanup_interval = interval(self.config.cleanup_interval);
76        let mut heartbeat_interval = interval(self.config.heartbeat_interval);
77        let mut message_buffer = Vec::new();
78
79        self.initialize().await?;
80
81        loop {
82            // Try to receive messages first
83            if let Err(e) = self.receive_messages(&mut message_buffer).await {
84                if !e.to_string().contains("No messages available") {
85                    error!("Error receiving messages: {}", e);
86                }
87            } else {
88                for (from, message) in message_buffer.drain(..) {
89                    if let Err(e) = self.handle_message(from, message).await {
90                        error!("Error handling message from {}: {}", from, e);
91                    }
92                }
93            }
94
95            tokio::select! {
96                // Handle incoming commands
97                command_opt = self.command_rx.recv() => {
98                    if let Some(command) = command_opt {
99                        if let Err(e) = self.handle_command(command).await {
100                            error!("Error handling command: {}", e);
101                        }
102                    } else {
103                        // Channel closed, exit loop
104                        break Ok(());
105                    }
106                }
107
108                // Cleanup old state
109                _ = cleanup_interval.tick() => {
110                    self.cleanup_old_state().await;
111                }
112
113                // Send heartbeats
114                _ = heartbeat_interval.tick() => {
115                    if let Err(e) = self.send_heartbeat().await {
116                        warn!("Failed to send heartbeat: {}", e);
117                    }
118                }
119
120                // Prevent busy waiting
121                _ = tokio::time::sleep(Duration::from_millis(1)) => {}
122            }
123        }
124    }

126    async fn initialize(&mut self) -> Result<()> {
127        // Try to restore state from persistence
128        if let Some(persisted_state) = self.persistence.load_state().await? {
129            info!("Restoring state from persistence");
130            
131            // Restore engine state
132            self.engine_state.current_phase.store(
133                persisted_state.current_phase.value(),
134                std::sync::atomic::Ordering::Release,
135            );
136            self.engine_state.last_committed_phase.store(
137                persisted_state.last_committed_phase.value(),
138                std::sync::atomic::Ordering::Release,
139            );
140
141            // Restore state machine if snapshot exists
142            if let Some(snapshot) = persisted_state.snapshot {
143                let mut sm = self.state_machine.lock().await;
144                sm.restore_snapshot(&snapshot).await?;
145            }
146        }
147
148        // Initialize network connections
149        let connected_nodes = self.network.lock().await.get_connected_nodes().await?;
150        self.engine_state.update_active_nodes(connected_nodes);
151
152        info!("Engine initialized successfully");
153        Ok(())
154    }

156    async fn handle_command(&mut self, command: EngineCommand) -> Result<()> {
157        match command {
158            EngineCommand::ProcessBatch(request) => {
159                self.process_batch_request(request).await
160            }
161            EngineCommand::Shutdown => {
162                info!("Shutting down consensus engine");
163                return Err(RabiaError::internal("Shutdown requested"));
164            }
165            EngineCommand::ForcePhaseAdvance => {
166                self.advance_to_next_phase().await
167            }
168            EngineCommand::TriggerSync => {
169                self.initiate_sync().await
170            }
171            EngineCommand::GetStatistics(tx) => {
172                let stats = self.engine_state.get_statistics();
173                let _ = tx.send(stats);
174                Ok(())
175            }
176        }

179    async fn process_batch_request(&mut self, request: CommandRequest) -> Result<()> {
180        if !self.engine_state.has_quorum() {
181            let _ = request.response_tx.send(Err(RabiaError::QuorumNotAvailable {
182                current: self.engine_state.get_active_nodes().len(),
183                required: self.engine_state.quorum_size,
184            }));
185            return Ok(());
186        }
187
188        // Add batch to pending
189        let batch_id = self.engine_state.add_pending_batch(request.batch.clone(), self.node_id);
190        
191        // Start consensus for this batch
192        self.propose_batch(batch_id, request.batch).await?;
193
194        // Note: The response will be sent when consensus completes
195        // For now, we'll store the response channel in the pending batch
196        Ok(())
197    }

199    async fn propose_batch(&mut self, batch_id: BatchId, batch: CommandBatch) -> Result<()> {
200        let phase_id = self.engine_state.advance_phase();
201        
202        debug!("Proposing batch {} in phase {}", batch_id, phase_id);
203
204        // Randomly choose initial value (key aspect of Rabia protocol)
205        let initial_value = if self.rng.gen_bool(0.5) {
206            StateValue::V0
207        } else {
208            StateValue::V1
209        };
210
211        // Update phase with proposal
212        self.engine_state.update_phase(phase_id, |phase| {
213            phase.batch_id = Some(batch_id);
214            phase.proposed_value = Some(initial_value.clone());
215            phase.batch = Some(batch.clone());
216        })?;
217
218        // Broadcast proposal
219        let proposal = ProposeMessage {
220            phase_id,
221            batch_id,
222            value: initial_value,
223            batch: Some(batch),
224        };
225
226        let message = ProtocolMessage::propose(self.node_id, proposal);
227        self.network.lock().await.broadcast(message, Some(self.node_id)).await?;
228
229        Ok(())
230    }

232    async fn handle_message(&mut self, from: NodeId, message: ProtocolMessage) -> Result<()> {
233        // Validate incoming message
234        if let Err(e) = message.validate() {
235            warn!("Received invalid message from {}: {}", from, e);
236            return Err(e);
237        }
238
239        // Validate message source
240        if message.from != from {
241            warn!("Message claims to be from {} but received from {}", message.from, from);
242            return Err(RabiaError::network("Message source mismatch"));
243        }
244
245        match message.message_type {
246            MessageType::Propose(propose) => {
247                self.handle_propose(from, propose).await
248            }
249            MessageType::VoteRound1(vote) => {
250                self.handle_vote_round1(from, vote).await
251            }
252            MessageType::VoteRound2(vote) => {
253                self.handle_vote_round2(from, vote).await
254            }
255            MessageType::Decision(decision) => {
256                self.handle_decision(from, decision).await
257            }
258            MessageType::SyncRequest(request) => {
259                self.handle_sync_request(from, request).await
260            }
261            MessageType::SyncResponse(response) => {
262                self.handle_sync_response(from, response).await
263            }
264            MessageType::NewBatch(new_batch) => {
265                self.handle_new_batch(from, new_batch).await
266            }
267            MessageType::HeartBeat(heartbeat) => {
268                self.handle_heartbeat(from, heartbeat).await
269            }
270            MessageType::QuorumNotification(_) => {
271                // Handle quorum notifications
272                Ok(())
273            }
274        }
275    }

277    async fn handle_propose(&mut self, from: NodeId, propose: ProposeMessage) -> Result<()> {
278        if !self.engine_state.has_quorum() {
279            return Ok(()); // Ignore proposals when no quorum
280        }
281
282        debug!("Received proposal from {} for phase {}", from, propose.phase_id);
283
284        // Store the batch if we don't have it
285        if let Some(batch) = &propose.batch {
286            self.engine_state.add_pending_batch(batch.clone(), from);
287        }
288
289        // Determine our vote for round 1
290        let vote = self.determine_round1_vote(&propose).await;
291
292        // Update phase data
293        self.engine_state.update_phase(propose.phase_id, |phase| {
294            phase.batch_id = Some(propose.batch_id);
295            if phase.proposed_value.is_none() {
296                phase.proposed_value = Some(propose.value);
297            }
298            if phase.batch.is_none() {
299                phase.batch = propose.batch.clone();
300            }
301        })?;
302
303        // Send round 1 vote
304        let vote_msg = VoteRound1Message {
305            phase_id: propose.phase_id,
306            batch_id: propose.batch_id,
307            vote,
308            voter_id: self.node_id,
309        };
310
311        let message = ProtocolMessage::vote_round1(self.node_id, from, vote_msg);
312        self.network.lock().await.send_to(from, message).await?;
313
314        Ok(())
315    }

317    async fn determine_round1_vote(&mut self, propose: &ProposeMessage) -> StateValue {
318        // Rabia's voting strategy: nodes vote based on their local state and randomization
319        // This implements the Rabia protocol's voting rules for round 1
320        
321        // Check if we have any conflicting proposals for this phase
322        let phase = self.engine_state.get_phase(&propose.phase_id);
323        
324        match phase {
325            Some(existing_phase) => {
326                // If we already have a proposal for this phase
327                if let Some(existing_value) = &existing_phase.proposed_value {
328                    if *existing_value == propose.value {
329                        // Same proposal - vote for it
330                        propose.value.clone()
331                    } else {
332                        // Conflicting proposal - vote ? (uncertain)
333                        StateValue::VQuestion
334                    }
335                } else {
336                    // First time seeing this phase - randomized vote
337                    self.randomized_vote(&propose.value)
338                }
339            }
340            None => {
341                // New phase - randomized vote based on Rabia protocol
342                self.randomized_vote(&propose.value)
343            }
344        }
345    }

347    fn randomized_vote(&mut self, proposed_value: &StateValue) -> StateValue {
348        // Rabia's randomized voting: bias towards V1 for liveness
349        // while maintaining safety through randomization
350        match proposed_value {
351            StateValue::V0 => {
352                // For V0 proposals, vote V0 with probability 0.5, else VQuestion
353                if self.rng.gen_bool(0.5) {
354                    StateValue::V0
355                } else {
356                    StateValue::VQuestion
357                }
358            }
359            StateValue::V1 => {
360                // For V1 proposals, vote V1 with higher probability for liveness
361                if self.rng.gen_bool(0.6) {
362                    StateValue::V1
363                } else {
364                    StateValue::VQuestion
365                }
366            }
367            StateValue::VQuestion => {
368                // For uncertain proposals, default to VQuestion
369                StateValue::VQuestion
370            }
371        }
372    }

374    async fn handle_vote_round1(&mut self, from: NodeId, vote: VoteRound1Message) -> Result<()> {
375        debug!("Received round 1 vote from {} for phase {}", from, vote.phase_id);
376
377        // Update phase with vote
378        self.engine_state.update_phase(vote.phase_id, |phase| {
379            phase.add_round1_vote(from, vote.vote);
380        })?;
381
382        // Check if we have enough votes to proceed to round 2
383        if let Some(phase) = self.engine_state.get_phase(&vote.phase_id) {
384            if let Some(majority_vote) = phase.has_round1_majority(self.engine_state.quorum_size) {
385                self.proceed_to_round2(vote.phase_id, majority_vote, phase.round1_votes).await?;
386            }
387        }
388
389        Ok(())
390    }

392    async fn proceed_to_round2(
393        &mut self,
394        phase_id: PhaseId,
395        round1_result: StateValue,
396        round1_votes: std::collections::HashMap<NodeId, StateValue>,
397    ) -> Result<()> {
398        debug!("Proceeding to round 2 for phase {} with result {:?}", phase_id, round1_result);
399
400        // Rabia protocol round 2 voting rules
401        let round2_vote = match round1_result {
402            StateValue::V0 => {
403                // Round 1 decided V0 - must vote V0 for safety
404                StateValue::V0
405            }
406            StateValue::V1 => {
407                // Round 1 decided V1 - must vote V1 for safety
408                StateValue::V1
409            }
410            StateValue::VQuestion => {
411                // Round 1 was inconclusive - Rabia's randomized choice
412                // Bias towards V1 for liveness while maintaining safety
413                self.determine_round2_vote_for_question(&round1_votes)
414            }
415        };
416
417        // Update our phase with round 2 vote
418        self.engine_state.update_phase(phase_id, |phase| {
419            phase.add_round2_vote(self.node_id, round2_vote.clone());
420        })?;
421
422        // Broadcast round 2 vote
423        let vote_msg = VoteRound2Message {
424            phase_id,
425            batch_id: self.engine_state.get_phase(&phase_id)
426                .and_then(|p| p.batch_id)
427                .unwrap_or_else(BatchId::new),
428            vote: round2_vote,
429            voter_id: self.node_id,
430            round1_votes,
431        };
432
433        let message = ProtocolMessage::vote_round2(self.node_id, self.node_id, vote_msg);
434        self.network.lock().await.broadcast(message, Some(self.node_id)).await?;
435
436        Ok(())
437    }

439    fn determine_round2_vote_for_question(
440        &mut self,
441        round1_votes: &std::collections::HashMap<NodeId, StateValue>,
442    ) -> StateValue {
443        // When round 1 is inconclusive, use Rabia's strategy:
444        // 1. Count the non-? votes to see if there's a preference
445        // 2. If tied or no clear preference, randomize with bias towards V1
446        
447        let v0_count = round1_votes.values().filter(|&v| *v == StateValue::V0).count();
448        let v1_count = round1_votes.values().filter(|&v| *v == StateValue::V1).count();
449        
450        if v1_count > v0_count {
451            // More V1 votes in round 1 - prefer V1
452            if self.rng.gen_bool(0.8) {
453                StateValue::V1
454            } else {
455                StateValue::V0
456            }
457        } else if v0_count > v1_count {
458            // More V0 votes in round 1 - prefer V0
459            if self.rng.gen_bool(0.7) {
460                StateValue::V0
461            } else {
462                StateValue::V1
463            }
464        } else {
465            // Tied or no clear preference - bias towards V1 for liveness
466            if self.rng.gen_bool(0.6) {
467                StateValue::V1
468            } else {
469                StateValue::V0
470            }
471        }
472    }

492    async fn make_decision(&mut self, phase_id: PhaseId, decision: StateValue) -> Result<()> {
493        info!("Decision reached for phase {}: {:?}", phase_id, decision);
494
495        // Update phase with decision
496        self.engine_state.update_phase(phase_id, |phase| {
497            phase.set_decision(decision.clone());
498        })?;
499
500        // Apply the batch if decision is V1 (commit)
501        if decision == StateValue::V1 {
502            if let Some(phase) = self.engine_state.get_phase(&phase_id) {
503                if let Some(batch) = &phase.batch {
504                    self.apply_batch(batch).await?;
505                    if let Err(e) = self.engine_state.commit_phase(phase_id) {
506                        error!("Failed to commit phase {}: {}", phase_id, e);
507                        return Err(e);
508                    }
509                }
510            }
511        }
512
513        // Broadcast decision
514        let phase = self.engine_state.get_phase(&phase_id).unwrap();
515        let decision_msg = DecisionMessage {
516            phase_id,
517            batch_id: phase.batch_id.unwrap_or_else(BatchId::new),
518            decision,
519            batch: phase.batch,
520        };
521
522        let message = ProtocolMessage::decision(self.node_id, decision_msg);
523        self.network.lock().await.broadcast(message, Some(self.node_id)).await?;
524
525        Ok(())
526    }

531        // Apply commands without holding the lock for too long

544    async fn handle_decision(&mut self, _from: NodeId, decision: DecisionMessage) -> Result<()> {
545        debug!("Received decision for phase {}: {:?}", decision.phase_id, decision.decision);
546
547        // Update our phase data with the decision
548        self.engine_state.update_phase(decision.phase_id, |phase| {
549            phase.set_decision(decision.decision.clone());
550            if phase.batch.is_none() {
551                phase.batch = decision.batch.clone();
552            }
553        })?;
554
555        // Apply the batch if we haven't already and decision is commit
556        if decision.decision == StateValue::V1 {
557            if let Some(batch) = &decision.batch {
558                // Check if we've already applied this batch
559                let last_committed = self.engine_state.last_committed_phase();
560                if decision.phase_id > last_committed {
561                    self.apply_batch(batch).await?;
562                    if let Err(e) = self.engine_state.commit_phase(decision.phase_id) {
563                        error!("Failed to commit phase {} from decision: {}", decision.phase_id, e);
564                        return Err(e);
565                    }
566                }
567            }
568        }
569
570        Ok(())
571    }

617    async fn resolve_sync(&mut self, responses: std::collections::HashMap<NodeId, SyncResponseMessage>) -> Result<()> {
618        info!("Resolving sync with {} responses", responses.len());
619
620        // Find the most recent state among responses
621        let latest_response = responses
622            .values()
623            .max_by_key(|r| r.responder_phase.value())
624            .cloned();
625
626        if let Some(latest) = latest_response {
627            let current_phase = self.engine_state.current_phase();
628            
629            if latest.responder_phase > current_phase {
630                info!("Syncing to phase {} from phase {}", latest.responder_phase, current_phase);
631                
632                // Update our phase
633                self.engine_state.current_phase.store(
634                    latest.responder_phase.value(),
635                    std::sync::atomic::Ordering::Release,
636                );
637
638                // Restore state machine if snapshot provided
639                if let Some(snapshot) = latest.state_snapshot {
640                    let mut sm = self.state_machine.lock().await;
641                    sm.restore_snapshot(&snapshot).await?;
642                }
643            }
644        }
645
646        // Clear sync responses
647        self.engine_state.clear_sync_responses();
648        Ok(())
649    }

678
679        self.network.lock().await.broadcast(message, Some(self.node_id)).await?;
680        Ok(())
681    }
682
683    async fn advance_to_next_phase(&mut self) -> Result<()> {
684        let new_phase = self.engine_state.advance_phase();
685        info!("Advanced to phase {}", new_phase);
686        Ok(())
687    }
688
689    async fn initiate_sync(&mut self) -> Result<()> {
690        info!("Initiating synchronization");
691
692        let request = SyncRequestMessage {
693            requester_phase: self.engine_state.current_phase(),
694            requester_state_version: self.engine_state.get_state_version(),
695        };
696
697        // Send sync request to all active nodes
698        let active_nodes = self.engine_state.get_active_nodes();
699        for node_id in active_nodes {
700            if node_id != self.node_id {
701                let message = ProtocolMessage::sync_request(self.node_id, node_id, request.clone());
702                self.network.lock().await.send_to(node_id, message).await?;
703            }
704        }
705
706        Ok(())
707    }
708
709    async fn cleanup_old_state(&mut self) {
710        let removed_phases = self.engine_state.cleanup_old_phases(self.config.max_phase_history);
711        let removed_batches = self.engine_state.cleanup_old_pending_batches(300); // 5 minutes
712
713        if removed_phases > 0 || removed_batches > 0 {
714            debug!("Cleaned up {} old phases and {} old batches", removed_phases, removed_batches);
715        }
716    }
717
718    async fn receive_messages(&self, buffer: &mut Vec<(NodeId, ProtocolMessage)>) -> Result<()> {
719        // Try to receive multiple messages in a batch for efficiency
720        let mut network = self.network.lock().await;
721        
722        match timeout(Duration::from_millis(10), network.receive()).await {
723            Ok(Ok((from, message))) => {
724                buffer.push((from, message));
725                
726                // Try to get more messages without blocking
727                for _ in 0..10 { // Limit to prevent starvation
728                    match timeout(Duration::from_millis(1), network.receive()).await {
729                        Ok(Ok((from, message))) => buffer.push((from, message)),
730                        _ => break,
731                    }
732                }
733            }
734            Ok(Err(e)) => return Err(e),
735            Err(_) => {
736                // Timeout - no messages available
737            }
738        }
739
740        Ok(())
741    }
742}
